┌───────────────────────────────────────────────────────────────────────────┐
│ 🗄️ DATABASE SCHEMA                                                        │
└───────────────────────────────────────────────────────────────────────────┘

  TABLE: doctor_availability
  ━━━━━━━━━━━━━━━━━━━━━━━━━━━
  id                UUID         Primary Key
  doctorId          UUID         FK → doctor_profiles.id (CASCADE DELETE)
  clinicId          UUID         FK → clinics.id (CASCADE DELETE)
  dayOfWeek         INTEGER      0 = Sunday, 6 = Saturday
  isActive          BOOLEAN      Day enabled/disabled
  startTime         TEXT         HH:MM format (24-hour)
  endTime           TEXT         HH:MM format (24-hour)
  slotDurationMin   INTEGER      Slot duration in minutes (5–120)
  maxPatients       INTEGER      Max patients for day (1–200)
  createdAt         TIMESTAMP    Record creation time
  updatedAt         TIMESTAMP    Last update time

  INDEXES:
  ━━━━━━━━
  PRIMARY KEY (id)
  UNIQUE (doctorId, clinicId, dayOfWeek)
  INDEX (doctorId)
  INDEX (clinicId)
  INDEX (dayOfWeek)
